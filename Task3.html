function sumOfEvenNumbers(arr) { let sum = 0; for (let i = 0; i < arr.length; i++) { if (arr[i] % 2 === 0) { sum += arr[i]; } } return sum; } // Test the function with the example array [1, 2, 3, 4, 5] const numbers = [1, 2, 3, 4, 5]; const result = sumOfEvenNumbers(numbers); console.log(result); // Output: 6
